    """
    –ü–æ–ª—É—á–∞–µ—Ç –≤—Å–µ –ø—Ä–µ–¥—Å—Ç–æ—è—â–∏–µ –∑–∞–ø–∏—Å–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –≤ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ–º –≤–∏–¥–µ.

            return []
        
        appointments_data = []
        for appointment in appointments:
            # –§–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º –¥–∞—Ç—É –∏ –≤—Ä–µ–º—è

            # –§–æ—Ä–º–∏—Ä—É–µ–º –¥–µ—Ç–∞–ª—å–Ω–æ–µ –æ–ø–∏—Å–∞–Ω–∏–µ
            details = f"{date_str} –≤ {time_str}: {appointment.service.name} –∫ –º–∞—Å—Ç–µ—Ä—É {appointment.master.name}"
            
            appointments_data.append({
                "id": appointment.id,
                "details": details
            })
        
        return appointments_data
    except Exception as e:
        return []
